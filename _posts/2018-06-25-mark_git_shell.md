	4.重新添加源代码库
	git remote add upstream <fork link>
	git remote -v

	5.从源仓库更新同步代码
	git fetch upstream

	6.合并到本地代码库
	git merge upstream/master

	7.拉取远程代码库
	git pull origin master

	8.更新远程代码库
	git push
